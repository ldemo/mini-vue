const Comp = {
	setup({ props, emits }) {
		return () => (
			<div
				onClick={() => emits('changeColor')}
				style={{ color: props.fontColor }}
			>
				hello world
			</div>
		)
	}
}
createApp({
	setup() {
		const fontColor = reactive({ value: '#5dbe8a' })
		const handleChangeColor = () => {
			fontColor.value = '#baccd9'
		}

		return () => (
			<Comp
				fontColor={fontColor}
				class="text-white"
				changeColor={handleChangeColor}
			/>
		)
	}
}).mount('#root')
