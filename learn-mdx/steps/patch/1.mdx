const Comp = {
	setup() {
		const msg = "I'm comp"
		return () => (
			<div>{msg}</div>
		)
	}
}

createApp({
	render() {
		return (
			<div id="demo">
				<span>hello world</span>
				<Comp />
			</div>
		)
	}
}).mount('#root')
